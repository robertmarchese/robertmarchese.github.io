				 function countBs(bString) {
        var count = 0;
        for (var i = 0; i < bString.length; i++) {
          if (bString.charAt(i) == 'B') {
            count = (count + 1);
          }
        }
        return count;
      }


      $("#q4button").on("click", function() {

                    function range(start, end) {

                var arr = [];
                if (start <= end) { // find out how to do an if NaN
                  for (start; start <= end; start++) {
                    arr.push(start);

                  }
                }
                return arr;
              }

                    function sum(array) {
                var sm = 0;
                for (var i = 0; i < array.length; i++) {
                  sm += parseInt(array[i]);
                }
                return sm;
              }

                    	function reverseArray(array) {
			var rev = []
			for (var i = array.length - 1; i >= 0; i--) {

				rev.push(array[i])
			}
			return rev
		}


		$("#q8Button").on("click", function () {

                     function arrayReverseInPlace(array) {
                for (var i = 0; i < array.length/2; i++) {
                  var temp = array[i];
                  array[i] = array[array.length - 1 - i];
                  array[array.length-1-i] = temp;
